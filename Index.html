utils/feature_extraction.py 
def extract_features(detections):
    if not detections:
        return {"nodule_count": 0, "avg_size": 0.0, "locations": []}
    count = len(detections)
    sizes = [d["coord"][3] for d in detections]
    avg_size = sum(sizes) / count
    locations = [d["coord"][:3] for d in detections]
    return {
        "nodule_count": count,
        "avg_size": round(avg_size, 2),
        "locations": locations
    }
digital_twin/digital_twin.py 
import json
from datetime import datetime

class DigitalTwin:
    def __init__(self, patient_id: str):
        self.patient_id = patient_id
        self.scan_history = []

    def add_scan(self, features: dict, date=None):
        if date is None:
            date = datetime.now().strftime("%Y-%m-%d")
        state = {
            "date": date,
            "nodule_count": features["nodule_count"],
            "avg_size_mm": features["avg_size"],
            "locations": features["locations"]
        }
        if self.scan_history:
            prev = self.scan_history[-1]
            progression = self._compute_progression(prev, state)
            state["progression"] = progression
        else:
            state["progression"] = "Initial baseline"
        self.scan_history.append(state)
        return state

    def _compute_progression(self, prev, current):
        count_delta = current["nodule_count"] - prev["nodule_count"]
        size_delta = current["avg_size_mm"] - prev["avg_size_mm"]
        if count_delta > 0 or size_delta > 2:
            return "Growing (↑)"
        elif count_delta < 0 or size_delta < -2:
            return "Shrinking (↓)"
        return "Stable"

    def get_history(self):
        return self.scan_history
llm/clinical_reasoning.py 
def generate_clinical_reasoning(state, history):
    if not state:
        return "No scan data yet."
    prog = state.get("progression", "Unknown")
    count = state["nodule_count"]
    size = state["avg_size_mm"]
    risk = "HIGH" if count > 2 or size > 12 else "MEDIUM" if count > 0 else "LOW"
    return f"""🧠 LLM Clinical Reasoning (Mock GPT-4o style)

Current Status: {count} nodules • Avg size {size} mm • {prog}
Risk Level: {risk}

Progression Analysis: The digital twin shows {prog.lower()} pattern compared to previous scan.
Recommendation: Follow-up CT in 3 months + pulmonology consult.
This explanation was generated by combining imaging features with medical knowledge (RAG-style)."""

import streamlit as st
from utils.inference import mock_detect
from utils.feature_extraction import extract_features
from digital_twin.digital_twin import DigitalTwin
from llm.clinical_reasoning import generate_clinical_reasoning
import pandas as pd

st.set_page_config(page_title="LUNATWIN", page_icon="🫁", layout="wide")
st.title("🫁 Predictive Medical Digital Twin")
st.caption("LUNA16 3D ViT + LLM Clinical Reasoning")

if "dt" not in st.session_state:
    st.session_state.dt = DigitalTwin("P001")

dt = st.session_state.dt

col1, col2 = st.columns([3, 2])
with col1:
    if st.button("📸 Simulate New CT Scan (LUNA16 Model)", type="primary", use_container_width=True):
        detections = mock_detect()
        features = extract_features(detections)
        new_state = dt.add_scan(features)
        st.success("✅ Full pipeline executed!")
        st.json(new_state)

with col2:
    st.subheader("Digital Twin History")
    if dt.scan_history:
        df = pd.DataFrame(dt.scan_history)
        st.dataframe(df, use_container_width=True)

st.subheader("LLM Explanation")
if dt.scan_history:
    reasoning = generate_clinical_reasoning(dt.scan_history[-1], dt.scan_history)
    st.info(reasoning)
else:
    st.info("Run a scan to generate clinical insights")
